### **Page Transition**
```tsx
import { motion } from 'framer-motion'

export default function Page() {
  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      exit={{ opacity: 0, y: -20 }}
      transition={{ duration: 0.4 }}
    >
      {/* Page content */}
    </motion.div>
  )
}
```

### **Card Reveal on Scroll**
```tsx
<motion.div
  initial={{ opacity: 0, y: 50 }}
  whileInView={{ opacity: 1, y: 0 }}
  viewport={{ once: true }}
  transition={{ duration: 0.6 }}
>
  <POICard {...poi} />
</motion.div>
```

**Secondary (Gold):** CTAs, badges, luxury elements
**Accent (Rose):** Highlights, personality fit, romantic elements
**Surface (Light Gray):** Cards, chat bubbles
